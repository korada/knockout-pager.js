Knockout-Pager.js

Plugin to incorporate paging to any grid built using knockout. This plugin doesn't generate html, but will provide all the methods to bind to any custom html.

How to add to your html?

  <script type="text/javascript" src="knockout-3.0.0.js"></script>
  <script type="text/javascript" src="knockout-pager.js"></script>

Usage


  //initialization
  var pager=ko.pager(10);
  //set items count
  pager.itemCount(20);
  //set Page Size
  pager.setPageSize(5);
  // set Number of pages in a frame
  pager.framePagesCount(5)
  //Current Page
  pager.currentPage();
  //Get Total Pages
  pager.totalPages();
  //Set page
  pager.setPage();
  //Go to Next page
  pager.nextPage();
  //Go to previous page
  pager.previousPage();
  //Current pages numbers in the frame
  pager.currentFramePagers();
  //Next Frame
  pager.nextFrame();
  //previous Frame
  pager.previousFrame();
  //Has next frame
  pager.hasNextFrame();
  //has previous frame
  pager.hasPreviousFrame();

Sample HTML


  <div class="pager" id="#koPager"> 
    <button data-bind="click:pager.firstPage"> <i class="fa fa-step-backward"></i> </button> 
    <button data-bind="click: pager.previousPage"> <i class="fa fa-caret-left"></i> </button> 
    <button data-bind="click: pager.previousFrame">...</button> 
    <!--ko foreach: pager.currentFramePages()--> 
    <button data-bind="text: $data, click: $parent.pager.setPage.bind($data), css: {'selected':$parent.pager.currentPage()==$data}"></button> 
    <!--/ko--> 
    <button data-bind="click: pager.nextFrame">...</button> 
    <button data-bind="click:pager.nextPage"> <i class="fa fa-caret-right"></i> </button> 
    <button data-bind="click:pager.lastPage"> <i class="fa fa-step-forward"></i> < /button> 
  </div>

JS


  
  var vm= function(){
    var self=this;
  self.pager=ko.pager(0);
  self.pager.frameSize(5);
  self.pager.itemCount(10);
  }

ko.applyBindings(new vm(),document.getElementById("koPager"));
